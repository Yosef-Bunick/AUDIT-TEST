    def compare_projects(root: Path, skip: list[str]) -> dict:
        """Discover subdirs, run audit-test on each, return scores."""
        results = {}
        for d in root.iterdir():
            if d.name in skip or not d.is_dir():
                continue
            # Run audit-test on this project
            out = subprocess.run(
                ["audit-test", "-p", str(d), "phd", "wiring", "high", "--json"],
                capture_output=True, text=True, timeout=60
            )
            data = json.loads(out.stdout)

            # Extract HIGH counts per module
            wiring_high = len(data.get("wiring", {}).get("findings", []))
            phd_high = len(data.get("phd", {}).get("findings", []))

            results[d.name] = {
                "wiring_high": wiring_high,
                "phd_high": phd_high,
                "total_high": wiring_high + phd_high,
                "file_count": len(list(d.rglob("*.py"))),
            }
        return results


    Agent calls: compare_projects("/path/to/root", ["makeup,bunch"]) → gets back a dict of project names and scores. One round-trip.



    Tool 2: benchmark_compilation(project_path)

    python
    def benchmark_compilation(project: Path) -> dict:
        """Time how fast this project compiles. Faster = lighter engine."""
        py_files = [f for f in project.rglob("*.py")
                    if "pycache" not in str(f) and ".venv" not in str(f)]

        loc = 0; funcs = 0; loops = 0
        t0 = time.monotonic()

        for f in py_files:
            tree = ast.parse(f.read_text(encoding="utf-8"))
            # Count structural elements
            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                    funcs += 1
                elif isinstance(node, (ast.For, ast.While)):
                    loops += 1
            # Count lines using end_lineno
            loc += max(getattr(n, "end_lineno", 0) or 0 for n in ast.walk(tree))

        elapsed = time.monotonic() - t0
        return {
            "compile_seconds": round(elapsed, 3),
            "loc": loc,
            "function_count": funcs,
            "loop_count": loops,
        }


    Agent gets: {"compile_seconds": 0.033, "loc": 2522, "function_count": 120, "loop_count": 34}. Now it knows 233am is 2.6x lighter than 106am.

    def extract_features(project: Path) -> dict:
        """Find what this project actually DOES — public API surface + feature keywords."""
        products = set()
        pipeline_stages = set()

        for f in project.rglob("*.py"):
            if "pycache" in str(f) or ".venv" in str(f):
                continue
            tree = ast.parse(f.read_text(encoding="utf-8"))

            for node in ast.walk(tree):
                # Extract function names as capability signals
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                    name = node.name
                    if name.startswith("_"):
                        continue
                    # Classify by naming convention
                    if "apply" in name or "render" in name or "process" in name or "draw" in name:
                        pipeline_stages.add(name)

                # Extract product/feature keywords from string literals
                if isinstance(node, ast.Constant) and isinstance(node.value, str):
                    val = node.value.lower()
                    # Known product categories
                    for kw in ["concealer", "contour", "blush", "highlight",
                              "eyeshadow", "eyeliner", "lip", "brow", "bronzer",
                              "foundation", "mascara", "powder", "3d", "uv", "pbr"]:
                        if kw in val:
                            products.add(kw)

        return {
            "products": sorted(products),
            "pipeline_stages": sorted(pipeline_stages),
            "product_count": len(products),
            "pipeline_count": len(pipeline_stages),
        }


    Agent gets: {"products": ["blush", "bronzer", "concealer", "contour", ...], "pipeline_stages": ["apply_makeup", "process_frame", "render_2p5d", ...]}. Now it can see 233am is MISSING eyeshadow_crease and render_2p5d compared to 106am.

    def analyze_architecture(project: Path) -> dict:
        """Extract function signatures and class structure to understand modularity."""
        functions = []
        classes = {}
        imports = set()

        for f in project.rglob("*.py"):
            if "__pycache__" in str(f) or ".venv" in str(f):
                continue
            tree = ast.parse(f.read_text(encoding="utf-8"))

            for node in ast.walk(tree):
                # Public functions = capability surface
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                    if not node.name.startswith("_"):
                        functions.append({
                            "name": node.name,
                            "file": str(f.relative_to(project)),
                            "line": node.lineno,
                            "args": len(node.args.args),
                        })

                # Classes = subsystems
                if isinstance(node, ast.ClassDef):
                    methods = [n.name for n in ast.walk(node)
                              if isinstance(n, (ast.FunctionDef, ast.AsyncFunctionDef))
                              and not n.name.startswith("_")]
                    classes[node.name] = methods

                # Imports = dependencies
                if isinstance(node, (ast.Import, ast.ImportFrom)):
                    for alias in node.names:
                        imports.add(alias.name.split(".")[0])

        # Classify architecture patterns
        has_pipeline = any("render" in f["name"] for f in functions)
        has_3d = any("3d" in f["name"].lower() for f in functions)
        has_segmentation = "segment" in {f["name"] for f in functions}
        has_smoothing = any("smooth" in f["name"] for f in functions)
        has_3view = any("view" in f["name"] or "fuse" in f["name"] for f in functions)
        is_monolith = "process_frame" in {f["name"] for f in functions} and len(functions) < 15

        return {
            "function_count": len(functions),
            "class_count": len(classes),
            "classes": classes,
            "imports": sorted(imports),
            "architecture": {
                "pipeline_separated": has_pipeline,
                "has_3d": has_3d,
                "has_skin_segmentation": has_segmentation,
                "has_face_smoothing": has_smoothing,
                "has_3view_capture": has_3view,
                "is_monolith": is_monolith,
            },
        }

    Tool 5: estimate_performance(project_path)

    Estimates runtime performance from code structure when actual
    execution isn't possible (missing deps, wrong OS, etc.).

    python
    def estimate_performance(project: Path) -> dict:
        """Analyze code for performance characteristics without running it."""
        compute_ops = 0
        buffer_sizes = []
        uses_roi = False
        resolution_dependent = False

        for f in project.rglob("*.py"):
            if "__pycache__" in str(f) or ".venv" in str(f):
                continue
            src = f.read_text(encoding="utf-8", errors="replace")

            # Count heavy operations
            for pattern in ["cv2.", "np.", "torch.", "tf.", "jax."]:
                compute_ops += src.count(pattern)

            # Find resolution/buffer constants
            import re
            for m in re.finditer(r"(?:SIZE|WIDTH|HEIGHT|RESOLUTION)\s*=\s*(\d+)", src):
                buffer_sizes.append(int(m.group(1)))

            # ROI processing = not full-frame
            if "ROI" in src or "roi" in src or "region_of_interest" in src:
                uses_roi = True

            # Full-frame = resolution-dependent speed
            if "shape[0]" in src or "shape[1]" in src or ".width" in src or ".height" in src:
                resolution_dependent = True

        return {
            "compute_ops": compute_ops,
            "min_buffer_size": min(buffer_sizes) if buffer_sizes else None,
            "uses_roi": uses_roi,
            "resolution_dependent": resolution_dependent,
            "estimated_speed": "fast" if uses_roi and buffer_sizes and min(buffer_sizes) <= 512
                        else "medium" if uses_roi or compute_ops < 200
                        else "slow",
        }


    Agent gets: {"compute_ops": 129, "min_buffer_size": 256, "uses_roi": True,
                 "resolution_dependent": False, "estimated_speed": "fast"}

    Combined with compile speed and architecture analysis, the judger now has
    a complete picture: code quality (audit-test), compile speed (Tool 2),
    features (Tool 3), architecture (Tool 4), and estimated runtime speed
    (Tool 5). It can say "233am is 2x faster at runtime because it uses
    a 256x256 buffer with ROI processing, but 3.28 is more accurate
    because of 2.5D depth-aware rendering."


    Tool 6: classify_dead_symbols(project_path, wiring_findings)

    Wiring says "X is dead" but doesn't say whether that matters.
    This tool checks: is dead code a harmless leftover, or missing functionality?

    python
    def classify_dead_symbols(project: Path, dead_symbols: list[dict]) -> dict:
        """Categorize dead symbols by whether they hurt the product."""
        critical = []    # missing feature — the code exists but doesn't run
        replaced = []    # old API — replaced by newer version (harmless)
        stubs = []       # utility never used — might be needed later

        for sym in dead_symbols:
            name = sym["name"]
            file_path = project / sym["file"]

            # Search ALL project files for references
            refs = []
            for f in project.rglob("*.py"):
                src = f.read_text(encoding="utf-8", errors="replace")
                if name in src and f != file_path:
                    refs.append(str(f.relative_to(project)))

            if refs:
                continue  # false positive — actually referenced

            # Classify by function name pattern
            if any(kw in name.lower() for kw in
                   ["render", "apply", "process", "depth", "create", "generate",
                    "detect", "segment", "track", "update", "draw"]):
                critical.append({
                    "name": name,
                    "reason": "Render/pipeline function defined but never called — missing feature",
                })
            elif any(kw in name.lower() for kw in
                     ["lerp", "clamp", "unit", "perp", "to_np", "dist",
                      "smoothstep", "offset", "gaussian"]):
                stubs.append({
                    "name": name,
                    "reason": "Math/utility stub — might be import-time or future use",
                })
            else:
                replaced.append({
                    "name": name,
                    "reason": "Likely replaced or callback — check manually",
                })

        return {
            "critical": critical,
            "stubs": stubs,
            "replaced": replaced,
            "accuracy_impact": len(critical) > 0,
            "summary": f"{len(critical)} dead features, {len(stubs)} dead stubs, "
                       f"{len(replaced)} replaced/callback"
        }


    Agent gets back:
    {
        "critical": [
            {"name": "render_2p5d", "reason": "2.5D renderer never called"},
            {"name": "create_depth_map", "reason": "Depth computation never called"},
            {"name": "apply_makeup", "reason": "Old 2D pipeline never called"}
        ],
        "stubs": [
            {"name": "lerp", "reason": "math utility"},
            {"name": "unit", "reason": "math utility"},
        ],
        "accuracy_impact": true,
        "summary": "3 dead features, 6 dead stubs, 3 replaced/callback"

    Surgeon v2: Cross-file copy/replace

    Current surgeon works on one file. For porting features between projects,
    it needs to copy code blocks from one file into another.

    def copy_lines(src: Path, start: int, end: int,
                   dest: Path, after_line: int) -> str:
        """Copy lines start..end from src, insert after after_line in dest."""
        src_lines = src.read_text(encoding="utf-8").splitlines(keepends=True)
        dest_lines = dest.read_text(encoding="utf-8").splitlines(keepends=True)

        # Extract source block
        block = src_lines[start - 1 : end]
        block_text = "".join(block)

        # Insert into destination
        dest_lines[after_line:after_line] = block
        dest.write_text("".join(dest_lines), encoding="utf-8")

        return f"copied {src}:{start}-{end} → {dest}:{after_line+1} ({len(block)} lines)"

    def port_feature(src_project: Path, dest_project: Path,
                     function_name: str) -> dict:
        """Find a function in src, copy it + its imports into dest.

        Steps:
        1. Find function definition + body in src
        2. Find any imports that function needs (not already in dest)
        3. Insert imports at top of dest, function at end of dest
        4. Run black on dest
        5. Return what was copied
        """
        # Find function in src
        for f in src_project.rglob("*.py"):
            tree = ast.parse(f.read_text(encoding="utf-8"))
            for node in ast.walk(tree):
                if isinstance(node, ast.FunctionDef) and node.name == function_name:
                    start = node.lineno
                    end = max(getattr(n, "end_lineno", start) for n in node.body) or start
                    block = f.read_text(encoding="utf-8").splitlines()[start-1:end]
                    return {
                        "function": function_name,
                        "file": str(f.relative_to(src_project)),
                        "lines": f"{start}-{end}",
                        "block": "".join(block),
                    }
        return {"error": f"function {function_name} not found"}

      surgeon copy src.py 750:800 dest.py 0          (insert at top)
      surgeon replace src.py 750:800 dest.py 15:20    (replace dest lines)
